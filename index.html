Cascade.min.JS

the easiest way to cascade animations


How to implement:

<style> // make your animation .div { --margin: 100px; --finalMargin: 40px; animation: fadeIn 0.5s ease-out forwards; margin-top: var(--margin); } @keyframes fadeIn { 0% { margin-top: var(--margin); opacity: 0; } 100% { margin: var(--finalMargin) opacity: 1; } } </style> // add the minified js file <script src="js/cascade.min.js"></script> //declare your objects and call the function(); <script> let cascaded = document.querySelectorAll('.div'); let obj = new Cascade(cascaded,0.1).cascade(); </script>
